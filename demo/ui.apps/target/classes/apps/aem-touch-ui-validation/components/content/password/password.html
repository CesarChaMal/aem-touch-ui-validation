Password validation sample

    <passwordA
        jcr:primaryType="nt:unstructured"
        sling:resourceType="granite/ui/components/foundation/form/password"
        fieldLabel="No validation"
        fieldDescription="No validation"
        name="./passwordA"/>
    <passwordB
        jcr:primaryType="nt:unstructured"
        sling:resourceType="granite/ui/components/foundation/form/password"
        fieldLabel="Required, Min & Max length"
        fieldDescription="Required, Min & Max length"
        required="{Boolean}true"
        minlength="{Long}6"
        maxlength="{Long}10"
        name="./passwordB"/>
    <passwordC
        jcr:primaryType="nt:unstructured"
        sling:resourceType="granite/ui/components/foundation/form/password"
        fieldLabel="Email"
        fieldDescription="Email"
        required="{Boolean}true"
        pattern="email"
        name="./passwordC"/>
    <passwordD
        jcr:primaryType="nt:unstructured"
        sling:resourceType="granite/ui/components/foundation/form/password"
        fieldLabel="Custom Regex"
        fieldDescription="Custom Regex for social security number must match ^\\d{3}-\\d{2}-\\d{4}$"
        required="{Boolean}true"
        pattern="^\\d{3}-\\d{2}-\\d{4}$"
        name="./passwordD"/>
    <passwordE
        jcr:primaryType="nt:unstructured"
        sling:resourceType="granite/ui/components/foundation/form/password"
        fieldLabel="URL with active validation"
        fieldDescription="Start typeing "http://www.adobe.com" and notice the validation happens on keypress rather than on blur."
        required="{Boolean}true"
        activeValidation="{Boolean}true"
        pattern="url"
        name="./passwordE"/>
